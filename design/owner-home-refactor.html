Cambio 01 · Hero
"Estás al día" reemplaza al $0

El estado en lenguaje natural responde la pregunta directa. El número grande pasa al "próximo vencimiento" — lo que sí necesita acción futura.

Cambio 02 · Stats
Próximo · Monto · Racha

Tres datos accionables: cuándo vence, cuánto, y feedback positivo (4 meses al día). La racha incentiva mantener el comportamiento.

Cambio 03 · CTA contextual
El botón cambia según el estado

Al día → "Adelantar próximo pago". Pendiente → "Subir comprobante" rojo. Aprobado este mes → "Ver mi pago". Un solo CTA, mensaje correcto.

Cambio 04 · Avisos con valor
Asambleas, cortes, novedades

Avisos con icono semántico, fecha relativa y dot de "no leído". Si el back devuelve vacío, mostrar próximo evento del calendario en su lugar.

Cambio 05 · Acciones secundarias
Historial + Reclamo accesibles

Dos accesos directos antes del fold: ver historial completo y crear reclamo de mantenimiento. Sin sumar peso al hero.

// ─── Owner Home · render dinámico ────────────────────────────

export async function renderOwnerHome(root) {
  root.innerHTML = skeleton('owner-home');

  const [me, summary, news] = await Promise.all([
    apiCall('/api/me'),
    apiCall('/api/owner/summary'),     // estado + próximo
    apiCall('/api/news?limit=3'),
  ]);

  root.innerHTML = tpl({ me, summary, news });
  bindEvents(root, { me, summary });
}

// ─── Estado derivado ─────────────────────────────────────────

function deriveStatus(s) {
  if (s.overdue > 0)            return { kind: 'overdue',  label: 'Tenés deuda',    tone: 'red'    };
  if (s.pendingReceipt)         return { kind: 'pending',  label: 'Pago en revisión', tone: 'yellow' };
  if (s.paidThisMonth)          return { kind: 'paid',     label: 'Estás al día',    tone: 'green'  };
  return                              { kind: 'due',      label: 'Próximo a vencer', tone: 'green'  };
}

function deriveCTA(status) {
  switch (status.kind) {
    case 'overdue': return { label: 'Regularizar deuda',    action: 'pay-overdue',  tone: 'red'    };
    case 'pending': return { label: 'Ver mi comprobante',    action: 'view-pending', tone: 'yellow' };
    case 'paid':    return { label: 'Adelantar próximo pago', action: 'pay-next',     tone: 'green'  };
    default:           return { label: 'Subir comprobante',     action: 'upload',       tone: 'green'  };
  }
}

// ─── Template ────────────────────────────────────────────────

function tpl({ me, summary, news }) {
  const status = deriveStatus(summary);
  const cta    = deriveCTA(status);
  const nextIn = daysUntil(summary.nextDueDate);

  return `
    <header class="greet">
      <p class="hello">${greeting()} · ${fmtShortDate()}</p>
      <h1>Hola, <em>${me.firstName}</em> 👋</h1>
      <p class="meta">${me.barrio} · Lote ${me.lote}</p>
    </header>

    <section class="hero hero--${status.tone}">
      <div class="hero__top">
        <span class="period">${fmtMonth(summary.period)}</span>
        <span class="pill pill--${status.tone}">${status.label}</span>
      </div>
      <h2 class="hero__big">${heroLine(status, summary)}</h2>
      <p class="hero__desc">${heroDesc(status, summary)}</p>
      <ul class="hero__stats">
        <li><span>PRÓXIMO</span><b>${fmtShort(summary.nextDueDate)}</b><em>en ${nextIn} días</em></li>
        <li><span>MONTO</span><b>${money(summary.amount)}</b><em>expensa fija</em></li>
        <li><span>RACHA</span><b class="g">${summary.streak} meses</b><em>al día ✓</em></li>
      </ul>
    </section>

    <button class="cta cta--${cta.tone}" data-action="${cta.action}">
      <span class="cta__t">${cta.label}</span>
    </button>

    <div class="quick">
      <button data-action="history">Mi historial · ${summary.payments} pagos</button>
      <button data-action="claim">Nuevo reclamo</button>
    </div>

    ${summary.lastPayment ? tplLastPayment(summary.lastPayment) : ''}
    ${tplNews(news)}
  `;
}

// ─── Bindings ────────────────────────────────────────────────

function bindEvents(root, ctx) {
  root.querySelectorAll('[data-action]').forEach(btn => {
    btn.addEventListener('click', () => {
      switch (btn.dataset.action) {
        case 'upload':
        case 'pay-next':
        case 'pay-overdue':
          openModal('upload-receipt', { period: ctx.summary.period });
          break;
        case 'view-pending':
          openModal('receipt-detail', { id: ctx.summary.pendingId });
          break;
        case 'history':  navigate('/owner/history'); break;
        case 'claim':    openModal('new-claim');     break;
      }
    });
  });
}

// ─── Helpers de copy ─────────────────────────────────────────

function heroLine(s, d) {
  if (s.kind === 'paid')    return `Estás <em>al día</em><br>con tu expensa`;
  if (s.kind === 'pending') return `Tu pago está<br><em>en revisión</em>`;
  if (s.kind === 'overdue') return `Tenés <em>${money(d.overdue)}</em><br>pendientes`;
  return `Vence en <em>${daysUntil(d.nextDueDate)} días</em>`;
}

function heroDesc(s, d) {
  if (s.kind === 'paid')    return `Ya pagaste <b>${money(d.amount)}</b> el <b>${fmtShort(d.lastPayment.date)}</b>.`;
  if (s.kind === 'pending') return `Subiste el comprobante el <b>${fmtShort(d.pendingDate)}</b>. Te avisamos cuando se apruebe.`;
  if (s.kind === 'overdue') return `Llevás <b>${d.overdueMonths} meses</b> sin pago aprobado. Regularizá para evitar interés.`;
  return `Vence el <b>${fmtShort(d.nextDueDate)}</b>. Subí el comprobante cuando estés listo.`;
}
